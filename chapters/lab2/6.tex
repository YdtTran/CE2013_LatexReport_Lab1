\subsubsection{Theory calculations}
\textbf{\textit{Approximation}}: Diodes have $V_f = 0.78V$

% itemize them with bullet points
\begin{itemize}
    \item Minimum Value of $V_{R1} = 0$ (Because during negative half cycle, diode is off).
    \item Maximum Value of $V_{R1} = V_{\sin{Max}} - V_f = 22 - 0.78 = 21.22V$
    \item Duration (millisecond) for a cycle of $V_{R1}$ = $T_{cycle} = \dfrac{1}{FREQ} = \dfrac{1}{50} = 0.02s = 20ms$
\end{itemize}

\subsubsection{PSpice simulation}
% itemize them with bullet points

\begin{itemize}
    \item Minimum Value of $V_{R1} \approx -6.399018344997e-28V \approx 0V$.
    \item Maximum Value of $V_{R1} \approx 2.125892105879e+01V \approx 21.26V$.
    \item Duration (millisecond) for a cycle of $V_{R1} = 20ms$.
\end{itemize}
